# 手写 bind 的实现

call apply bind 的区别

相同点： 在 js 中 call、apply、bind 都是改变 this 的指向
不同点：1. call 和 apply 是改变 this 的指向同时立即执行函数, bind 是改变 this 的指向，返回一个函数，需要再次调用才能指向

  1. bind 和 call 参数是依次传进去， apply 参数是一个数组
var a = {
  name: "a",
  from: function(age, sex) {
    console.log(`${this.name}:年龄${age},性别${sex}`);
  },
};
var b = { name: "小米" };

a.from.bind(b, 29, "女")(); // 小米:年龄29,性别女
a.from.bind(b, 29)("男");
a.from.call(b, 29, "女"); // 小米:年龄29,性别女
a.from.apply(b, [29, "女"]);

# 实现 parseInt 方法

function _parseInt(s, radix) {
  s = s.trim();
  radix = radix || 10;
  var reg = /^([-\+]?)(\d+)/;
  const match = s.match(reg);
  if (!match) return NaN;
  const str = match[2];
  const len = str.length;
  let num = 0;
  for (let i = 0; i < len; i++) {
    console.log(typeof str[i]);
    let n = str[i].charCodeAt() - 48;
    if (n >= radix) return NaN;
    num += n * Math.pow(radix, len - 1 - i);
  }
  return match[1] === "-" ? `-${num}` : num;
}

# 实现函数防抖

防抖就是在一段时间内，无论触发了多少次回调都只触发最后一次 实现原理就是定时器， 每触发一次回调，原来的定时器清空， 定时器重新计时

/**
 * @param {Function} fn
 * @param {Number} wait 每次触发的等待时间
 */
function debounce(fn, wait) {
  var timer = null;
  return function(...arg) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, arg);
    }, wait);
  };
}
setInterval(() => {
  console.log("触发回调");
}, 2000);

// 保证time时间内有一次执行，最后一次也能执行
function debonce(fn, time){
    let timer = null
    let start = null
    return function(...argus){
        let current = new Date()
        if (!start) start = current
        if (current - start >= time) {
            fn.apply(this, argus)
            start = current
        }
        if (timer) clearTimeout(timer)
        timer = setTimeout(() => {
            let timeoutTime = new Date()
            if (timeoutTime - start >= time) {
                fn.apply(this, argus);
                start = timeoutTime
            }
        }, time)
    }
}
let fun = debonce(() => {
    console.log('回调')
}, 200)

setInterval(() => {
  fun(1)  
}, 100);

# 函数节流 throttle

防止函数被频繁调用 使用场景： 页面滚动、窗口大小变化、鼠标移动等

/**
 * @param {Function} fn 回调函数
 * @param {Number} wait 每隔wait时长才会触发一次回调
 */
var throttle = function(fn, wait) {
  var previewTime = null;
  return function(argumrnts) {
    var nowTime = new Date();
    if (!previewTime) previewTime = nowTime;
    if (nowTime - previewTime >= wait) {
      fn.apply(this, ...argumrnts);
      previewTime = nowTime;
    }
  };
};
const eventFn = throttle(() => {
  console.log(new Date());
}, 1000);

setInterval(eventFn, 50);

function createNew() {
  // 取出构造函数, shift 会修改原数组，所以 arguments 会被去除第一个参数
  var Constructor = [].shift.call(arguments);
  // 创建空对象
  var obj = {};
  obj.__proto__ = Object.create(Constructor.prototype);
  // 绑定 this 实现继承，obj 可以访问到构造函数中的属性
  var ret = Constructor.apply(obj, arguments);

  // 这一步是为了处理构造函数有返回值的情况
  return typeof ret === "object" ? ret : obj;
}

function _setTimeout(fn, wait, ...args) {
  var start = new Date();
  var timer;
  function loop() {
    var now = new Date();
    if (now - start >= wait) {
      fn.call(this, ...args);
      window.cancelAnimationFrame(timer);
    } else {
      timer = window.requestAnimationFrame(loop);
    }
  }

  window.requestAnimationFrame(loop);
}

# 二分查找实现findIndex

function findIndex(arr, target) {
    let left = 0;
    let right = arr.length - 1
    // let middle =
    let index = -1
    while(left <= right) {
        let middle = Math.floor((right-left) / 2) + left
        // console.log(right, left, middle)
        if (arr[middle] < target) {
            left = middle + 1
        } else if(arr[middle] > target) {
            right = middle - 1
        } else {
            index = middle
            right = middle - 1
        }
    }
    return index
}

# 实现promise.finally

finally方法本质上是then方法的特例

promise
.finally(() => {
  // 语句
});

// 等同于
promise
.then(
  result => {
    // 语句
    return result;
  },
  error => {
    // 语句
    throw error;
  }
);

Promise.prototype._finally = function (cb) {
  cb = typeof cb === "function" ? cb : () => {};

  const P = this.constructor; // 获取当前实例的构造函数


  // promise 成功走resolve 
  let onFufilled = function (data) {
    return P.resolve(cb()).then(() => {
      return data;
    });
  };

  // promise 失败走reject   err其实就是reject(err) 的err参数  finally最后还要返回这个err参数 
  let onReject = function (err) {
    return P.resolve(cb()).then(() => err);
  };

  // 等同于是then方法
  return this.then(onFufilled, onReject);
};
// ------------ 测试 --------------
const p = new Promise((resolve, reject) => {
  console.info("starting...");

    Math.random() > 0.5 ? resolve("success") : reject("fail");

});

})

# 实现promise.race

promise.race 是将多个Promise实例，包装成一个新的 Promise 实例

const p = Promise.race([p1, p2, p3])

上面代码中，只要p1, p2, p3 之中有一个实例率先改变状态， p的状态就会改变，那个率先改变的Promise实例的返回值就会传递给p的回调函数 Promise.race()方法的参数与Promise.all()方法一样，如果不是 Promise 实例，就会先调用下面讲到的Promise.resolve()方法，将参数转为 Promise 实例，再进一步处理。

const _race = (p) => {
  return new Promise((resolve, reject) => {
    p.forEach(item => {
      // Promise.resolve(item)是将item先转为 Promise 实例
      Promise.resolve(item).then(resolve, reject)
    })
  })
}

# Promise.all

 Promise._all = function (list) {
  return new Promise((resolve, reject) => {
    let len = list.length;
    let n = 0;
    let result = [];
    for (let i = 0; i < len; i++) {
      list[i]
        .then((data) => {
          n++;
          result[i] = data;
          if (n === len) resolve(result);
        })
        .catch((err) => {
          return reject(err);
        });
    }
  });
};

    let len = arr.length
    let result = []

# 观察者模式

// 观察者
class Objserver {
    constructor(fn) {
        this.cb = fn
    }

    update(params){
      this.cb(params)
    }
}

// 被观察者
class Subject {
    constructor(params) {
        this.observer = []
        this.params = params
    }

    addObserver(observer) {
        this.observer.push(observer)
    }

    notify() {
        let _this = this
        this.observer.forEach(fn => {
            fn.update(this.params)
        })
    }
}

const observer1 = new Objserver((params) => {
    console.log('我被通知更新了', params)
})

const subject = new Subject('被观察者的参数')
subject.addObserver(observer1)
subject.notify()
// 我被通知更新了 被观察者的参数

# 两个栈实现一个队列

// 封装一个栈，只有进栈，出栈操作。
class Stack {
    constructor(arr){
        this.ary = arr||[];
        this.length = this.ary.length;
    }
    push(item){
        this.ary.push(item);
        this.length++;
        // console.log(this.ary);
    }
    pop(){
        this.length--;
        return this.ary.pop();
    }

}

//两个栈模拟队列。

function stackToQueue(){
    this.stack1 = new Stack();
    this.stack2 = new Stack();
}
stackToQueue.prototype.enqueue = function(item){
    this.stack1.push(item);
}
stackToQueue.prototype.dequeue = function(){
    if(this.stack2.length){
        return this.stack2.pop();
    }else{
        while(this.stack1.length){
            this.stack2.push(this.stack1.pop());
        };
        return this.stack2.pop();
    }
}
