# 手写promise

function myPromise(executor){
    this.status = 'pending' // 状态 resolved rejected
    this.value = '' // 返回值
    this.err = ''
    this.successFn = [] // 成功回调
    this.errFn = []  // 失败回调

    const resolve = (value) =>  {
        this.status = 'resolved'
        this.successFn.forEach(fn => {
        setTimeout(() => {
            fn && fn(value)
        }, 0)
        });
    }

    const reject = (err) => {
        this.status === 'rejected'
        this.errFn.forEach(fn => {
            setTimeout(() => {
                fn && fn(err)
            }, 0)
        });
    }

    // promise执行出错
    try {
        executor(resolve, reject)
    } catch (err) {
        reject(err) 
    }
}
function handle(promise2,x,resolve,reject) {
    // 规范 2.3.1，x 不能和 promise2 相同，避免循环引用
    if (promise2 === x) {
       return reject(new TypeError('Chaining cycle detected for promise'))
    }
    // 上一步返回结果是个promise
    if (x instanceof myPromise) {
        if (x.status === 'pending') {
            x.then(function (value) {
                // 再次调用该函数是为了确认 x resolve 的
                // 参数是什么类型，如果是基本类型就再次 resolve
                // 把值传给下个 then
                handle(promise2, value, resolve, reject)
            }, reject)
        } else {
            x.then(resolve, reject)
        }
        
    } else if (x !== null && (typeof x === 'object' || typeof x === 'function')) {
        // 规范 2.3.3，判断 x 是否为对象或函数
        // 规范 2.3.3.2，如果不能取出 then，就 reject
        let then  = x.then
        if (typeof then === 'function') {
            then.call(
                x,
                y => { // 规范 2.3.3.3.1
                    if (called) return // 规范 2.3.3.3.3，即这三处谁先执行就以谁的结果为准
                    called = true
                    // 规范 2.3.3.3.1
                    return handle(promise2, y, resolve, reject)
                },
                r => {
                    if (called) return // 规范 2.3.3.3.3，即这三处谁先执行就以谁的结果为准
                    called = true
                    return reject(r)
                }
            )
        } else {
            resolve(x)
        }
    }else {
        resolve(x)
    }
}
myPromise.prototype.then = function (onResolve, onReject) {
    let _this = this
    if (typeof onResolve !== 'function') {
        onResolve = function (){}
    } 
    if (typeof onReject !== 'function') {
        onReject = function (){}
    }

    let promise2 = new myPromise((resolve, reject) => {
        if (_this.status === 'resolved') {
            try {
                let x = _this.onResolve(_this.value)
                handle(promise2, x, resolve, reject)
            } catch (err){
                reject(err)
            }
        } else
        if (_this.status === 'pending') {
            _this.successFn.push(
               () => {
                    try {
                        let x = _this.onResolve(_this.value)
                        handle(promise2, x, resolve, reject)
                    } catch (err){
                        reject(err)
                    }
               }
            )
            _this.errFn.push(
               () => {
                    try {
                        let x = _this.onRejected(_this.value)
                        handle(promise2, x, resolve, reject)
                    } catch (err){
                        reject(err)
                    }
               }
            )
        } else {
            try {
                let x = _this.onRejected(_this.value)
                handle(promise2, x, resolve, reject)
            } catch (err){
                reject(err)
            }
        }
    })
}

# Promise.all

Promise._all = function (list) {
  return new Promise((resolve, reject) => {
    let len = list.length;
    let n = 0;
    let result = [];
    for (let i = 0; i < len; i++) {
      list[i]
        .then((data) => {
          n++;
          result[i] = data;
          if (n === len) resolve(result);
        })
        .catch((err) => {
          return reject(err);
        });
    }
  });
};

# promise.allSettled

const rejectHandler = reason => ({status: "rejected", reason})
const resolveHandler = value => ({status: "fulfilled", value})
Promise.allSettled = promises =>
Promise.all(
    promises.map((promise) =>
    Promise.resolve(promise) 
        .then(resolveHandler, rejectHandler)
    )
    // 每个 promise 需要用 Promise.resolve 包裹下
    // 以防传递非 promise
);

// 使用
const p1 = Promise.resolve(1)
const p2 = Promise.resolve(2)
const p3 = new Promise((resolve, reject) => {
setTimeout(reject, 1000, 'three');
})
const promises = [p1, p2, p3]
Promise.allSettled(promises).then(console.log)

# 实现 promise.finally

finally 方法本质上是 then 方法的特例

promise.finally(() => {
  // 语句
});

// 等同于
promise.then(
  (result) => {
    // 语句
    return result;
  },
  (error) => {
    // 语句
    throw error;
  }
);

Promise.prototype._finally = function (callback) {
  let P = this.constructor;
  return this.then(
    value => P.resolve(callback()).then(() => value),
    reason => P.resolve(callback()).then(() => { throw reason })
  );
};
Promise.resolve(2)._finally(() => { console.log('999') }).then(res => { console.log(res) })

Promise.prototype._finally = function (cb) {
  cb = typeof cb === "function" ? cb : () => {};

  const P = this.constructor; // 获取当前实例的构造函数

  // promise 成功走resolve
  let onFufilled = function (data) {
    return P.resolve(cb()).then(() => {
      return data;
    });
  };

  // promise 失败走reject   err其实就是reject(err) 的err参数  finally最后还要返回这个err参数
  let onReject = function (err) {
    return P.resolve(cb()).then(() => err);
  };

  // 等同于是then方法
  return this.then(onFufilled, onReject);
};
// ------------ 测试 --------------
const p = new Promise((resolve, reject) => {
  console.info("starting...");

  setTimeout(() => {
    Math.random() > 0.5 ? resolve("success") : reject("fail");
  }, 1000);
});

// 正常顺序测试
p.then((data) => {
  console.log(`%c resolve: ${data}`, "color: green");
})
  .catch((err) => {
    console.log(`%c catch: ${err}`, "color: red");
  })
  ._finally(() => {
    console.info("finally: completed");
  });

// finally 前置测试
p._finally(() => {
  console.info("finally: completed");
})
  .then((data) => {
    //  返回的是原来resolve的值
    console.log(`%c resolve: ${data}`, "color: green");
  })
  .catch((err) => {
    //  返回的是原来reject的值
    console.log(`%c catch: ${err}`, "color: red");
  });

# 实现 promise.race

promise.race 是将多个 Promise 实例，包装成一个新的 Promise 实例

const p = Promise.race([p1, p2, p3]);

上面代码中，只要 p1, p2, p3 之中有一个实例率先改变状态， p 的状态就会改变，那个率先改变的 Promise 实例的返回值就会传递给 p 的回调函数 Promise.race()方法的参数与 Promise.all()方法一样，如果不是 Promise 实例，就会先调用下面讲到的 Promise.resolve()方法，将参数转为 Promise 实例，再进一步处理。

const _race = (p) => {
  return new Promise((resolve, reject) => {
    p.forEach((item) => {
      // Promise.resolve(item)是将item先转为 Promise 实例
      Promise.resolve(item).then(res=> {resolve(res)}, err => {reject(err)});
    });
  });
};

# async Await的实现

async await是Generator函数的语法糖 Generator 函数，可以通过 yield 关键字，把函数的执行流挂起，通过next()方法可以切换到下一个状态 Generator函数 next方法可以传递一个参数，让yield具有返回值

function fetch (p) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve(p)
        }, 1000)
    })
}
function* it() {
    let val1 = yield fetch(1)
    let val2 = yield fetch(2)
    let val3 = yield fetch(3)
    return val1 + val2 + val3
}
function co(it) {
    return new Promise((resolve, reject) => {
        function next(data) {
            let {done, value} = it.next(data)
            if (done) {
                resolve(value)
            } else {
                Promise.resolve(value).then(data => {next(data)}, err => throw Error())
            }
        }
        next()
    })
}

co(it()).then(data => {
    console.log(data);
  });
