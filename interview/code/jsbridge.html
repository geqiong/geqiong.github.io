# 封装jsBridge

native提供一个nativeOp功能， nativeOp是一个一步操作， 当操作完成时要通知js, 考虑超时功能

function myJsBridge() {
  return new Promise((resolve, reject) => {
    let time = new Date
    nativeOp()
    let T = setInterver(() => {
      let newTime = new Date()
      if (newTime - time > 100000) {
        T = null
        reject('time out)
      }
    }, 1000)
    window.onSuccess = function (res) {
      T = null
      resolve(res)
    }
  })
}
