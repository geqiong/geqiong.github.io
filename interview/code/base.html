# 1 数组的处理

已知一个数组 arr = [1,2,3,,4]

  • 数组的长度为多少？ arr.length = 5
  • 去除 arr 中的空数组项
// 用filter filter回调的2个参数为 value, index 返回新数组
arr.filter((value) => true); // [1,2,3,4]

// 用es6 flat api,数组扁平化 可以去除空数组项； 也是返回新数组
arr.flat();
  • arr2 = arr.map(parseInt)，请写下 arr2 的内容
    答案： [1, NaN, NaN, , NaN]
    原因： map 的回调接收三个参数 value index arr;
    parseInt 接收 2 个参数，第一个参数是 value, 第二个参数是 value 是几进制数；
    map 的遍历 相当于执行 parseInt(1, 0) parseInt(2, 1) parseInt(3, 2)
  • arr1 = [1,2,3] arr2 = [4,5,6] 合并两个数组
arr3 = arr1.concat(arr2);
arr3 = [...arr1, ...arr2];
  • 数组去重
Array.form(new Set(arr)) // Array.from 将类数组变为数组
[...new Set()]

# 2 字符串处理

replace 第二个参数是回调函数， 函数接收的参数为（matchStr, groups, index, sourceStr） matchStr 为整个正则匹配到的结果 groups 是正则分组匹配的结果 如果有多个分组就是 group1 group2 index 是匹配到的下标位置 sourceStr 是被匹配的原字符串

  • 1 写一个名为 toCamelCase 的方法，实现把类似'abc-def-ghi'的字符转换成'abcDefGhi'
function toCamelCase(str) {
  return str.replace(/-(.)/g, function($0, $1) {
    console.log($0, $1);
    return $1.toUpperCase();
  });
}
  • 2 写一个名为toDashJoin的方法，实现把驼峰形式字符串'abcDefGhi'转换成'abc-def-ghi'
function toDashJoin(str){
    return str.replace(/[A-Z]/g,function($0){
        return '-'+$0.toLowerCase()
    })
}
  • 3 写一个名为toCapitalize的方法，实现首字母大写功能（原来字母就是大写的不处理），如'i like css'转换成'I Like Css'。
function toCapitalize(str){
    return str.replace(/(\s+|^)(\w)/g,function($0,$1,$2){
        return $1+$2.toUpperCase()
    })
}
  • 4 写一个名为toBetterUrl的方法，实现把类似'CSS value type'转换成'css-value-type'（只需考虑空格和大小写处理）
function toBetterUrl(str){
    return str.replace(/[A-Z]/g,function($0){
        return $0.toLowerCase()
    }).replace(/\s+/g,'-')
}

# 给高优运算符加括号

const str = '11+2-3*4+5/2*4+10/5 - 12*3/20000*2'

function fn (str) {
    let res = ''
    let hasClose = false // 括号需要闭合
    for (let i = 0, len = str.length; i < len; i++) {
        let match = str.substring(i).match(/\d+(?=[\*|\/])/)

        if (match && match.length &&  match.index === 0) {
            if (!hasClose) {
                res = res + '(' + str[i]
                hasClose = true
            } else {
                res = res + str[i]
            }
        } else if (hasClose && (str[i+1] === '+' || str[i+1] === '-')){
            res = res + str[i] + ')'
            hasClose = false
        } else {
            res = res + str[i]
        }

    }
    if (hasClose) res = res + ')'
    return res
}

console.log(fn(str)) // 11+2-(3*4)+(5/2*4)+(10/5 )- (12*3/20000*2)
