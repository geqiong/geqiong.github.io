# 寻找对称数

比如 101 121 1001 ...

function find(start, end) {
    let len1 = (start + '').length
    let len2 = (end + '').length
    let isOdd = !!(Math.floor( len1 % 2))
    result = []
    for(i= Math.floor(len1 / 2); i<= Math.floor(len2 / 2); i++) {
        num = Math.pow(10, i-1)
        while(num < Math.pow(10, i)) {
            let right = Number(num).toString().split('').reverse().join('')
            if (isOdd) {
                let j = 0
                while(j < 10) {
                    let res = Number('' + num + j + right)
                    if (res >= start && res <= end) result.push(res)
                    j++
                    if (res > end) return result
                }
            } else {
                res = Number(num + right)
                if (res >= start && res <= end) result.push(res)
                if (res > end) return result
            }
            num++
        }
        if (isOdd) {
            isOdd = false
        } else {
            i--
            isOdd = true 
        }
    }
    return result
}
console.log(find(100, 1000))
