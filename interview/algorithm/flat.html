flat方法接收一个参数 n ; n 是扁平化几层， 默认一层

Array.prototype._flat = function (n = 1) {
    const arr = this
    if (n < 1 && n !== Infinity) return arr
    let len = arr.length
    let result = []
    for(let i = 0; i < len; i++) {
        if (arr[i] instanceof Array) {
            const a = n === Infinity || n >= 1 ? arr[i]._flat(n-1) : arr[i]
            result = result.concat(a)
        } else {
            result.push(arr[i])
        }
    }
    return result
}
console.log([0,[1,[2,[3,[4,[5]]]]]]._flat(Infinity))

// 递归的方式 展开所有层
function flat(arr) {
    let len = arr.length
    let result = []
    for(let i = 0; i < len; i++) {
        if (arr[i] instanceof Array) {
            result = result.concat(flat(arr[i]))
        } else {
            result.push(arr[i]) 
        }
    }
    return result
}
