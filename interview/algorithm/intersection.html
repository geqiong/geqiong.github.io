# 合并对象中有交集的数组

function fn(arr) {
  for (let i = 0; i < arr.length; i++) {
    let j = arr.length - 1
    while (i < j) {
      if (isInter(arr[i], arr[j])) {
        arr[i] = {
          start: Math.min(arr[i].start, arr[j].start),
          end: Math.max(arr[i].end, arr[j].end)
        }
        arr.splice(j, 1)
      }
      j--
    }
  }
  return arr
}

arr = [
  { start: 1, end: 6 },
  { start: 13, end: 19 },
  { start: 4, end: 12 },
  { start: 2, end: 7 },
  { start: 3, end: 8 },
  { start: 14, end: 20 },
]
console.log(fn(arr)) // [ { start: 1, end: 12 }, { start: 13, end: 20 } ]

先对数组进行start排序， 如果相邻的两个没有交集。跟后面的也不会有交集
原因就是i+1的start大于i的end i+1之后的start更是大于i的end

function fn(arr) {
  arr = arr.sort((a, b) => a.start - b.start)
  for (let i = 0; i < arr.length - 1; i++) {
    if (isInter(arr[i], arr[i + 1])) {
      arr[i] = {
        start: Math.min(arr[i].start, arr[i + 1].start),
        end: Math.max(arr[i].end, arr[i + 1].end)
      }
      arr.splice(i + 1, 1)
      i--
    }
  }
  return arr
}

arr = [
  { start: 1, end: 6 },
  { start: 13, end: 19 },
  { start: 4, end: 12 },
  { start: 2, end: 7 },
  { start: 3, end: 8 },
  { start: 14, end: 20 },
]
console.log(fn(arr))

# 合并数组

输入 [[1,3], [2,6], [7, 10]] 输出 [[1,6], [7,10]]

function checkIsInter(arr1, arr2) {
 if (arr1[arr1.length - 1] < arr2[0] || arr2[arr2.length - 1] < arr1[0]) {
     return false
 }
 return true
}

function mergeArr(arr) {
    arr = arr.sort((a, b) => a[0] - b[0])
    let i = 0
    while (i < arr.length - 1) {
        if (checkIsInter(arr[i], arr[i+1])) {
            let len_1 = arr[i].length - 1
            let len_2 = arr[i+1].length - 1
            let start = Math.min(arr[i][0], arr[i+1][0])
            let end = Math.max(arr[i][len_1], arr[i+1][len_2])
            arr[i] = [start, end] 
            arr.splice(i+1, 1)
        } else {
            i++
        }
    }
    return arr
}

console.log(mergeArr([[1, 3, 4], [2, 6, 9],[5, 8], [6, 10], [12, 14], [15,16]]))
