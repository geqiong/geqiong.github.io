# 求数组最大嵌套深度

function getArrMaxDepth(arr, level) {
    let max = level
    for (let i = 0; i < arr.length; i++) {
        let curentDeep = getArrMaxDepth(arr[i], level+1)
        max = Math.max(curentDeep, max)
    }
    return max
}
arr = [1,[2,[3, [4]]], 4, 5]
console.log(getArrMaxDepth(arr, 0))
